class Solution {
public:
    bool isAnagram(string s, string t) {

        if(s.size() != t.size())return false;

        vector<int>check(26, 0);
        for(int i=0; i<s.size(); i++){
            check[ s[i]-'a' ]++; 
        }
        for(char x : t ){
            if( check[x-'a']==0 ){
                return false;
            }
            --check[x-'a'];
        }

        return true;
    }
};
