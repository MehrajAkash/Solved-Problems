class Solution {
public:
    bool isPalindrome(string s) {
        
        string tmp="", str="";
        for(int i=0; i<s.size(); i++){
            if( (s[i]>='a' && s[i]<='z') || (s[i]>='A' && s[i]<='Z') || (s[i]>='0' && s[i]<='9') ){
                tmp += tolower(s[i]);
                str += tolower(s[i]);
            }
        }

        reverse(tmp.begin(), tmp.end());

        if(tmp == str){
            return true;
        }

        return false;
    }
};
